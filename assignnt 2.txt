3. Write a program to swap values of two int variables.
Ans:-
#include<stdio.h>
int main()
{
    int a,b,c;
    printf("Enter a two number ");
    scanf("%d%d",&a,&b);
    printf("a=%d b=%d \n",a,b);
    c=a;
    a=b;
    b=c;
    printf("\na=%d b=%d ",a,b);
	return 0;
}

7. Write a program to find the position of first 1 in LSB.
ans:-
#include<stdio.h>
int main()
{
    int x=8,count=0;
    int result = 0;
	
	while ( x !=0)
	{
	result =x&1;
	count++;
		
	if (result==1)
	{
		printf("%d",count);
		break;
		
		}
	x = x>>1;
}
		
	return 0;
}
