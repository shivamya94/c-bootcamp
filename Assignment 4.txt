3. Write a program to print the first 10 natural numbers in reverse order
Ans:#include<stdio.h>
int main()
{
    printf("The first 10 natural Numbers in Reverse\n");
    for(int i=10;i>=1;i--)
    {
        printf("%d\n",i);
    }
    return 0;
}
